  "query": string,             // Original user query
  "plan": ResearchPlan,           // Generated research plan
  "sources": Source[],           // Discovered sources
  "findings": Finding[],         // Accumulated findings
  "final_report": Report,        // Synthesized output
  "status": string               // Session status
}
